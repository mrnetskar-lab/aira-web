======================================================================

This patch only adds the minimum AI-side continuity logic needed so the UI can later show more believable behavior:
- availability
- active app context
- delayed follow-up queue
- simple busy / later behavior
- continuity snapshot in API result

This patch is intentionally small.

1) NEW SYSTEM FILE
======================================================================

CREATE:
`src/engine/systems/RelationshipContinuitySystem.js`

CREATE WITH:

```js
const DEFAULT_APP = {
  type: 'chat',
  mode: 'group',
  visibility: 'public'
};

const PROFILES = {
  Lucy: {
    prefers: ['evening', 'morning'],
    sleepDuring: ['sleep_hours'],
    busyBias: 0.22,
    followUpBias: 0.62,
    practicalBias: 0.20
  },
  Sam: {
    prefers: ['day', 'evening'],
    sleepDuring: ['sleep_hours'],
    busyBias: 0.42,
    followUpBias: 0.32,
    practicalBias: 0.56
  },
  Angie: {
    prefers: ['evening', 'late_night'],
    sleepDuring: ['sleep_hours'],
    busyBias: 0.16,
    followUpBias: 0.58,
    practicalBias: 0.18
  }
};

export class RelationshipContinuitySystem {
  update({ state, context, input }) {
    if (!state.continuity) {
      state.continuity = createDefaultContinuityState();
    }

    state.continuity.turnClock += 1;
    state.continuity.timeBlock = getTimeBlock(state.continuity.turnClock);
    state.continuity.activeApp = normalizeActiveApp(
      context?.active_app || state.continuity.activeApp || DEFAULT_APP
    );

    for (const [name, profile] of Object.entries(PROFILES)) {
      const availability = state.continuity.availability[name] || createAvailabilityState();
      const timeBlock = state.continuity.timeBlock;

      availability.sleeping = profile.sleepDuring.includes(timeBlock);
      availability.busy =
        !availability.sleeping &&
        timeBlock !== 'late_night' &&
        Math.random() < profile.busyBias;

      availability.online =
        !availability.sleeping &&
        (profile.prefers.includes(timeBlock) || Math.random() < 0.45);

      availability.seen = false;
      availability.replying = false;

      state.continuity.availability[name] = availability;
    }
  }

  markExposure({ state, input }) {
    if (!state.continuity) return;

    const lower = String(input || '').toLowerCase();

    for (const name of Object.keys(state.continuity.availability)) {
      const availability = state.continuity.availability[name];
      if (!availability || availability.sleeping || !availability.online) continue;

      const mentioned = lower.includes(name.toLowerCase());
      const rel = state.relationships?.[name] || {};
      const trust = normalize01(rel.trust, 0.5);
      const attraction = normalize01(rel.attraction, 0.3);
      const attachment = normalize01(rel.attachment, 0.3);
      const resentment = normalize01(rel.resentment, 0);
      const pride = normalize01(rel.pride, 0.3);

      const interestScore =
        trust * 0.25 +
        attraction * 0.25 +
        attachment * 0.25 +
        (1 - resentment) * 0.15 +
        (1 - pride) * 0.10;

      const seenChance = mentioned ? 0.82 : 0.35 + interestScore * 0.35;

      if (Math.random() < seenChance) {
        availability.seen = true;
        availability.lastSeenTurn = state.continuity.turnClock;
      }
    }
  }

  decideImmediateMode({ state, character }) {
    const availability = state.continuity?.availability?.[character];
    if (!availability) return 'normal';
    if (availability.sleeping) return 'none';
    if (availability.busy) return Math.random() < 0.55 ? 'practical' : 'later';
    if (availability.online) return Math.random() < 0.75 ? 'normal' : 'later';
    return 'later';
  }

  queueFollowUp({ state, character, type, delayTurns = 2, reason = 'continuity' }) {
    if (!state.continuity) return;

    const id = `evt_${character}_${state.continuity.turnClock}_${Math.floor(Math.random() * 10000)}`;

    state.continuity.queuedEvents.push({
      id,
      character,
      type,
      reason,
      createdTurn: state.continuity.turnClock,
      readyTurn: state.continuity.turnClock + delayTurns,
      consumed: false
    });
  }

  consumeReadyEvents(state) {
    if (!state.continuity) return [];

    const ready = [];
    for (const event of state.continuity.queuedEvents) {
      if (!event.consumed && event.readyTurn <= state.continuity.turnClock) {
        event.consumed = true;
        ready.push(event);
      }
    }

    state.continuity.queuedEvents = state.continuity.queuedEvents.filter((e) => !e.consumed);

    return ready.slice(0, 1).map((event) => ({
      character: event.character,
      type: event.type,
      text: buildFollowUpText(event.character, event.type)
    }));
  }

  getSnapshot(state) {
    const continuity = state.continuity || createDefaultContinuityState();

    return {
      timeBlock: continuity.timeBlock,
      activeApp: continuity.activeApp,
      availability: continuity.availability
    };
  }
}

function createDefaultContinuityState() {
  return {
    turnClock: 0,
    timeBlock: 'evening',
    activeApp: { ...DEFAULT_APP },
    availability: {
      Lucy: createAvailabilityState(),
      Sam: createAvailabilityState(),
      Angie: createAvailabilityState()
    },
    queuedEvents: []
  };
}

function createAvailabilityState() {
  return {
    online: true,
    sleeping: false,
    busy: false,
    seen: false,
    replying: false,
    lastSeenTurn: null
  };
}

function getTimeBlock(turnClock) {
  const slot = turnClock % 18;
  if (slot <= 3) return 'evening';
  if (slot <= 6) return 'late_night';
  if (slot <= 9) return 'sleep_hours';
  if (slot <= 13) return 'morning';
  return 'day';
}

function normalizeActiveApp(value) {
  return {
    type: value?.type || 'chat',
    mode: value?.mode || 'group',
    visibility: value?.visibility || 'public'
  };
}

function normalize01(value, fallback) {
  if (typeof value !== 'number') return fallback;
  if (value > 1) return Math.max(0, Math.min(1, value / 100));
  return Math.max(0, Math.min(1, value));
}

function buildFollowUpText(character, type) {
  const map = {
    Lucy: {
      soft_check_in: 'hey. i was thinking about what you said.',
      busy_reply: 'sorry. i was busy.',
      late_follow_up: 'still here.'
    },
    Sam: {
      soft_check_in: 'read it.',
      busy_reply: 'at work. later.',
      late_follow_up: 'answering now.'
    },
    Angie: {
      soft_check_in: 'wow okay. hi again :)',
      busy_reply: 'busy. text me after?',
      late_follow_up: 'you vanished on me.'
    }
  };

  return map[character]?.[type] || 'later.';
}
```

======================================================================

TARGET:
`src/engine/brain/AiraBrainController.js`

```js
const immediateMode = this.continuity?.decideImmediateMode({
  state,
  character: primary.name
```

- if `none` => skip immediate reply for that character
- if `practical` => force responseMode to `brief`
