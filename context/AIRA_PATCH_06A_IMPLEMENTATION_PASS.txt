2) CLIENT — INLINE SUBTEXT HELPERS
======================================================================

TARGET
`client/app.js`

ADD THESE HELPERS near the existing escape helpers:

```js
function escapeHtmlAllowInline(value) {
  return String(value)
    .replaceAll("&", "&amp;")
    .replaceAll("<", "&lt;")
    .replaceAll(">", "&gt;")
    .replaceAll('"', "&quot;")
    .replaceAll("'", "&#39;");
}

function cleanSubtext(thought) {
  const raw = String(thought || "").trim();
  if (!raw) return "";

  let text = raw;

  text = text.replace(/^\*+|\*+$/g, "");
  text = text.replace(/^[\(\[]+|[\)\]]+$/g, "");
  text = text.replace(/\s+/g, " ").trim();

  text = text.replace(/^(thinking|thought|internally|to myself|in my head)\s*[:\-]?\s*/i, "");
  text = text.replace(/^(i think|i feel|she thinks|he thinks)\s+/i, "");

  if (text.length > 90) {
    text = text.slice(0, 87).trimEnd() + "...";
  }

  return text;
}

function buildInlineSubtext(spoken, thought, meta = {}) {
  const safeSpoken = escapeHtmlAllowInline(spoken || "");
  if (!thought) return safeSpoken;

  const thoughtMode = meta?.thoughtMode || "hidden";
  if (thoughtMode === "hidden") return safeSpoken;

  const cleaned = cleanSubtext(thought);
  if (!cleaned) return safeSpoken;

  const strongClass = meta?.subtextStrength === "strong" ? " is-strong" : "";
  return `${safeSpoken} <span class="subtext-inline${strongClass}">${escapeHtmlAllowInline(cleaned)}</span>`;
}
```

7) CLIENT — ADD AMBIENT AUTOTALK RENDER LANE
======================================================================

TARGET
`client/app.js`

PURPOSE
Allow the frontend to slowly render room continuation if backend later sends:
```json
result.ambient.autoTalk
```

ADD near `wait(ms)`:
```js
let ambientToken = 0;
```

ADD helper:
```js
async function appendAmbientMessages(items) {
  const myToken = ++ambientToken;

  for (let i = 0; i < items.length; i += 1) {
    const delay = 1400 + Math.random() * 1200;
    await wait(delay);

    if (myToken !== ambientToken) return;
    if (state.isSending) return;

    const entry = items[i];
    state.messages.push({
      type: "character",
      agent: entry.agent || "AIRA",
      text: buildInlineSubtext(entry.spoken, entry.thought, entry.meta),
      thought: "",
      anomalous: false,
      toneClass: entry.meta?.toneClass || "",
      anomalyAware: Boolean(entry.meta?.anomalyAware),
      time: formatTime(),
    });

    renderMessages();
    saveLocalState();
  }
}
```

INTERRUPT RULE
At the start of `sendMessage(text)`, add:
```js
ambientToken += 1;
```

This cancels pending ambient lines when the player speaks again.

AFTER normal response rendering in `sendMessage(text)`, add:

```js
const ambientItems = payload?.result?.ambient?.autoTalk ?? [];
if (ambientItems.length) {
  appendAmbientMessages(ambientItems);
}
```

FAKE MODE NOTE
If the current room chat is still in local fake mode and not using `/api/ai/run`,
you may stub this field later.
The helper can still exist now safely.
